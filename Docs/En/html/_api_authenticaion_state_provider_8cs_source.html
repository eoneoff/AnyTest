97  private IEnumerable<Claim> ParseClaimsFromJwt(string jwt)
98  {
99  var claims = new List<Claim>();
100  var payload = jwt.Split('.')[1];//Gets paload from jwt token
101  var jsonBytes = ParseBase64WithoutPadding(payload);
102  var keyValuePairs = JsonSerializer.Deserialize <Dictionary<string, object>>(jsonBytes);
103 
104  keyValuePairs.TryGetValue(ClaimTypes.Role, out object roles);
105 
106  if(roles != null)
